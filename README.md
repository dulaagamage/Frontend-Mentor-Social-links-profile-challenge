```css
.list-item {
  display: block;
  width: 100%;
  text-align: center;
  padding: 0.7rem;
  border-radius: 0.3rem;
  background: var(--clr-neutral-Grey);
  transition: background 0.3s ease, color 0.3s ease;
}

  color: var(--clr-neutral-Off-Black);
  background: var(--clr-primary-Green);
```
